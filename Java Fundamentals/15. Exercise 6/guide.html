Print multiplication table for a provided number num 

public class Snippet { 
  public static void main(String args[]) {
    int num = 5;    
    //numRows specifies the number of rows in the multiplication table
    int numRows = 10;
        
    int result;
    //Run the loop from 0 to numRows
    for(int currentRow = 0 ; currentRow <= numRows ; currentRow++ ) {
      //Print the result in a below format.
      //5 * 1 = 5
      //5 * 2 = 10
      System.out.println(num+" * "+currentRow+" = "+(num*currentRow));
    }
  }
}
